The same file in a composition helper implementation using CycleJS (10.0.5): 
import xs from 'xstream'

export default (parent, child, ...dependencies) => sources => {
  let proxies = dependencies.reduce((proxies, dep) => ({
    ...proxies,
    [dep]: xs.create()
  }), {})
  const childsinks = child({...sources, ...proxies})
  const sinks = parent({...sources, 'childsinks': childsinks})
  Object.keys(proxies).forEach(proxy => proxies[proxy].imitate(sinks[proxy]))
  return sinks
}
The same file in an xstream, storage driver implementation using CycleJS (12.2.2): 
import xs from 'xstream'
const {localStorage} = window

const makeStoreDriver = (key) => (save$) => {
  save$.addListener({
    next: (i) => localStorage.setItem(key, i),
    error: () => {},
    complete: () => {}
  })
  return xs.of(localStorage.getItem(key))
}

export default makeStoreDriver
